/*Integer11°. Дано трехзначное число. Найти сумму и произведение его цифр. */

#include <stdio.h>

int main()
{
int A;
printf("A:"); // выводим сообщение A
scanf ("%d", &A); // вводим значения переменной A | %d — целое число типа int со знаком в десятичной системе счисления

printf("+:%d *:%d\n",A%10+(A%100)/10+A/100,(A%10)*((A%100)/10)*(A/100)); // ‘\n’ — перевод строки

return 0;
}



/* Begin11. Даны два ненулевых числа. Найти сумму, разность,
произведение и частное их модулей. */

#include <stdio.h>
#include <math.h>
int main() {

double a, b;
scanf("%lf%lf", &a, &b); //%lf — вещественный формат двойной точности (числа с плавающей точкой типа double)
double ma = fabs(a); // fabs(a) модуль вещественного числа
double mb = fabs(b); // fabs(b) модуль вещественного числа
printf("%.2f, %.2f, %.2f, %.2f\n", ma+mb, ma-mb, ma*mb, ma/mb); //%.2f печатаеn только первые 2 цифры после точки
return 0;
}


/* For11. Дано целое число N (> 0). Найти сумму N2 + (N + 1)2 + (N + 2)2 + … + (2·N)2(целое число). */

#include <stdio.h>
#include <math.h>

int main()
{
int n;
printf("N:"); // выводим сообщение N
scanf ("%d", &n); // вводим значения переменной n | %d — целое число типа int со знаком в десятичной системе счисления

int i, result=0;
for (i =0; i<=n; ++i) // цикл
result +=pow(n+i,2); // функция pow вычисляет x^y

printf("result = %d \n ",result); // ‘\n’ — перевод строки

return 0;
} 


/*Boolean11°. Даны два целых числа: A, B. Проверить истинность высказывания: «Числа A и B имеют одинаковую четность». */

#include <stdio.h>

int main()
{
int A,B;
printf("A:"); // выводим сообщение A
scanf ("%d", &A); // вводим значения переменной A | %d — целое число типа int со знаком в десятичной системе счисления

printf("B:"); // выводим сообщение B
scanf ("%d", &B); // вводим значения переменной B | %d — целое число типа int со знаком в десятичной системе счисления

printf("%d\n", ((A%2==0) & (B%2==0) )|((A%2!=0) & (B%2!=0) ) ); // ‘\n’ — перевод строки | '!='- равно
return 0;
}


/*While11°. Дано целое число N (> 1). Вывести наименьшее из целых чисел K, для которых сумма 1 + 2 + … + K будет больше или равна N, и саму эту сумму. */

#include <stdio.h>
int main()
{
int n;
printf("N:"); // выводим сообщение N
scanf ("%d", &n); // вводим значения переменной n | %d — целое число типа int со знаком в десятичной системе счисления
int k=0, temp=0;
while (temp<n) {
++k; // возвращает значение переменной
temp+=3;
}
printf("K: %d\nSum:%d\n",k,temp); // ‘\n’ — перевод строки
return 0;
}
